        $E(\boldsymbol{w}) = \frac{1}{2}||\boldsymbol{y}-\boldsymbol{t}||^2_2$  \\

        $\boldsymbol{y} = [y(x_1,\boldsymbol{w})...y(x_N,\boldsymbol{w})]^T$\\

\end{center}

\begin{equation*}
